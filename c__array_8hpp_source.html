169 void increase(uint8_t c, int64_t delta)
170 {
171 if (delta > 0)
172 {
173 bool b = this->occurs_in_effective_alphabet(c);
174 if (!b)
175 {
176 int64_t ins_pos = 0;
177 int64_t size = this->effective_alphabet.size();
178 for (int64_t i = 0; i < size; i++)
179 {
180 uint8_t p2 = i + 1 < size ? this->effective_alphabet[i + 1] : UINT8_MAX;
181 if (this->effective_alphabet[i] < c && c <= p2)
182 {
183 ins_pos = i + 1;
184 }
185 }
186 this->effective_alphabet.insert(this->effective_alphabet.begin() + ins_pos, c);
187 }
188 }
189
190 for (uint64_t i = c + 1; i < this->C.size(); i++)
191 {
192 this->C[i] += delta;
193 }
194
195 if (this->get_c_count(c) == 0)
196 {
197 int64_t id = this->get_c_id(c);
198 this->effective_alphabet.erase(this->effective_alphabet.begin() + id);
199 }
200 assert(this->verify());
201 }
