373 std::pair<uint64_t, uint64_t> nearest_search_by_text_position(uint64_t text_position) const
374 {
375 int64_t idx1 = this->circular_successor_index_query_on_sampled_isa(text_position);
376 int64_t idx2 = idx1 > 0 ? idx1 - 1 : this->sampled_isa_gap_vector.size() - 1;
377
378 uint64_t t1 = this->get_sampled_isa_value(idx1);
379 uint64_t t2 = this->get_sampled_isa_value(idx2);
380
381 uint64_t dist1 = t1 <= text_position ? text_position - t1 : t1 - text_position;
382 uint64_t dist2 = t2 <= text_position ? text_position - t2 : t2 - text_position;
383 if (dist1 <= dist2)
384 {
385 return std::pair<uint64_t, uint64_t>(idx1, t1);
386 }
387 else
388 {
389 return std::pair<uint64_t, uint64_t>(idx2, t2);

562 void insert(int64_t new_sa_index, int64_t new_sa_value)
563 {
564 assert(new_sa_index <= (int64_t)this->pom.size());
565
566 int64_t idx = this->sampled_isa_gap_vector.successor_index(new_sa_value);
567
568 if (idx == -1)
569 {
570 idx = this->sampled_isa_gap_vector.size();
571 assert(idx <= (int64_t)this->pom.size());
572
573 int64_t diff = new_sa_value - this->sampled_isa_gap_vector.psum();
574
575 this->sampled_isa_gap_vector.insert(idx, diff);
576
577 this->pom.insert(new_sa_index, idx);
578 }
579 else
580 {
581 assert(idx <= (int64_t)this->pom.size());
582
583 int64_t v = this->sampled_isa_gap_vector.psum(idx);
584 int64_t diff = this->sampled_isa_gap_vector.at(idx);
585 int64_t diff2 = v - new_sa_value;
586 int64_t diff1 = diff - diff2;
587 this->sampled_isa_gap_vector.remove(idx);
588 this->sampled_isa_gap_vector.insert(idx, diff1);
589 this->sampled_isa_gap_vector.insert(idx + 1, diff2);
590
591 this->pom.insert(new_sa_index, idx);
592 }
593
594 assert(this->verify());
595 }

610 this->sampled_isa_gap_vector.remove(isa_index);
611
612 this->sampled_isa_gap_vector.insert(isa_index, diff1 + diff2);
613
614 }
615 else
616 {
617 this->sampled_isa_gap_vector.remove(isa_index);
618
619 }
620
621 this->pom.erase(sa_index);
622 assert(this->verify());
623 }
