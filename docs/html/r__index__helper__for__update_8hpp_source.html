35 struct PositionInformation
36 {
37 uint64_t p = UINT64_MAX;
38 SAValue value_at_p = UINT64_MAX;
39 SAValue value_at_p_minus = UINT64_MAX;
40 SAValue value_at_p_plus = UINT64_MAX;
41
42 void update_for_deletion(uint64_t removed_position_on_text, SAIndex removed_position_on_sa, uint64_t upper_sa_value, uint64_t lower_sa_value)
43 {
44 if (this->value_at_p > removed_position_on_text)
45 {
46 this->value_at_p--;
47 }
48 if (removed_position_on_sa < this->p)
49 {
50 this->p--;
51 }
52 if (this->value_at_p_minus == removed_position_on_text)
53 {
54 this->value_at_p_minus = upper_sa_value;
55 }
56 if (this->value_at_p_minus > removed_position_on_text)
57 {
58 this->value_at_p_minus--;
59 }
60
61 if (this->value_at_p_plus == removed_position_on_text)
62 {
63 this->value_at_p_plus = lower_sa_value;
64 }
65 if (this->value_at_p_plus > removed_position_on_text)
66 {
67 this->value_at_p_plus--;
68 }
69 }
70 };
