282 {

284 int64_t idx2 = idx1 > 0 ? idx1 - 1 : this->sampled_isa_gap_vector.size() - 1;

290 uint64_t dist2 = t2 <= text_position ? text_position - t2 : t2 - text_position;

423 void insert(int64_t new_sa_index, int64_t new_sa_value)
424 {
425 assert(new_sa_index <= (int64_t)this->pom.size());
426
427 int64_t idx = this->sampled_isa_gap_vector.successor_index(new_sa_value);
428
429 if (idx == -1)
430 {
431 idx = this->sampled_isa_gap_vector.size();
432 assert(idx <= (int64_t)this->pom.size());
433
434 int64_t diff = new_sa_value - this->sampled_isa_gap_vector.psum();
435
436 this->sampled_isa_gap_vector.insert(idx, diff);
437 this->pom.insert(new_sa_index, idx);
438 }
439 else
440 {
441 assert(idx <= (int64_t)this->pom.size());
442
443 int64_t v = this->sampled_isa_gap_vector.psum(idx);
444 int64_t diff = this->sampled_isa_gap_vector.at(idx);
445 int64_t diff2 = v - new_sa_value;
446 int64_t diff1 = diff - diff2;
447 this->sampled_isa_gap_vector.remove(idx);
448 this->sampled_isa_gap_vector.insert(idx, diff1);
449 this->sampled_isa_gap_vector.insert(idx + 1, diff2);
450 this->pom.insert(new_sa_index, idx);
451 }
452
453 assert(this->verify());
454 }
