204 static DynamicRLBWT build_from_RLBWT(const std::vector<uint8_t> &_rlbwt_chars, const std::vector<uint64_t> &_rlbwt_runs, const std::vector<uint8_t> &alphabet, int message_paragraph = stool::Message::SHOW_MESSAGE)
205 {
206
207 if (message_paragraph >= 0 && _rlbwt_chars.size() > 0)
208 {
209 std::cout << stool::Message::get_paragraph_string(message_paragraph) << "Constructing Dynamic RLBWT from RLBWT..." << std::endl;
210 }
211 std::chrono::system_clock::time_point st1, st2;
212 st1 = std::chrono::system_clock::now();
213
214 stool::sequence::DynamicWaveletTree tmp_head_chars_of_RLBWT = stool::sequence::DynamicWaveletTree::build(_rlbwt_chars, alphabet);
215 stool::prefix_sum::VLCDequeSPSI tmp_run_length_vector = stool::prefix_sum::VLCDequeSPSI::build(_rlbwt_runs);
216
217 std::vector<uint64_t> c_counters;
218 std::vector<uint64_t> c_run_counters;
219 std::vector<uint64_t> c_run_counters2;
220
221 c_counters.resize(256, 0);
222 c_run_counters.resize(256, 0);
223
224 for (int64_t i = 0; i < (int64_t)_rlbwt_chars.size(); i++)
225 {
226 uint8_t c = _rlbwt_chars[i];
227 int64_t run = _rlbwt_runs[i];
228 c_counters[c] += run;
229 c_run_counters[c]++;
230 }
231 for (uint64_t v : c_run_counters)
232 {
233 if (v != 0)
234 {
235 c_run_counters2.push_back(v);
236 }
237 }
238
239 stool::fm_index::CArray tmp_cArray = stool::fm_index::CArray::build(c_counters);
240 stool::prefix_sum::VLCDequeSPSI tmp_c_run_counters = stool::prefix_sum::VLCDequeSPSI::build(c_run_counters2);
241
242 std::vector<uint64_t> indexes;
243 indexes.resize(_rlbwt_chars.size(), UINT64_MAX);
244 for (uint64_t i = 0; i < _rlbwt_chars.size(); i++)
245 {
246 indexes[i] = i;
247 }
248 std::sort(indexes.begin(), indexes.end(), [&](const int64_t &lhs, const int64_t &rhs)
249 {
250 if(_rlbwt_chars[lhs] != _rlbwt_chars[rhs]){
251 return _rlbwt_chars[lhs] < _rlbwt_chars[rhs];
252 }else{
253 return lhs < rhs;
254 } });
255 stool::prefix_sum::VLCDequeSPSI tmp_run_length_vector_sorted_by_F;
256 for (auto i : indexes)
257 {
258 tmp_run_length_vector_sorted_by_F.push_back(_rlbwt_runs[i]);
259 }
260 DynamicRLBWT r;
261 r.head_chars_of_RLBWT.swap(tmp_head_chars_of_RLBWT);
262 r.run_length_vector.swap(tmp_run_length_vector);
263 r.run_length_vector_sorted_by_F.swap(tmp_run_length_vector_sorted_by_F);
264 r.c_run_counters.swap(tmp_c_run_counters);
265 r.cArray.swap(tmp_cArray);
266 r._text_size = r.run_length_vector.psum();
267
268 st2 = std::chrono::system_clock::now();
269 if (message_paragraph >= 0 && _rlbwt_chars.size() > 0)
270 {
271 uint64_t sec_time = std::chrono::duration_cast<std::chrono::seconds>(st2 - st1).count();
272 uint64_t ms_time = std::chrono::duration_cast<std::chrono::milliseconds>(st2 - st1).count();
273 uint64_t per_time = ((double)ms_time / (double)r._text_size) * 1000000;
274
275 std::cout << stool::Message::get_paragraph_string(message_paragraph) << "[END] Elapsed Time: " << sec_time << " sec (" << per_time << " ms/MB)" << std::endl;
276 }
277
278 return r;
279 }

624 int64_t select(uint8_t c, int64_t nth) const
625 {
626 if (nth == 0)
627 {
628 return -1;
629 }
630 else
631 {
632 nth--;
633 int64_t c_rank = this->get_c_id(c);
634 if (c_rank == -1)
635 {
636 return -1;
637 }
638 else
639 {
640 uint64_t f_idx = this->get_f_index_by_char(c);
