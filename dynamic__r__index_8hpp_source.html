531 RunPosition backward_search_sub1(RunPosition rp, uint8_t c) const
532 {
533 uint8_t c_rp = this->dbwt.access_character_by_run_index(rp.run_index);
534
535 if (c == c_rp)
536 {
537 return rp;
538 }

540 {
541 uint64_t rank = this->dbwt.rank_on_first_characters_of_RLBWT(c, rp.run_index);
542 uint64_t count = this->dbwt.c_run_count(c);
543 if (rank < count)
544 {
545 uint64_t y = this->dbwt.select_on_first_characters_of_RLBWT(rank, c);
546 return RunPosition(y, 0);
547 }
548 else
549 {
550 return RunPosition::create_null_value();
551 }
552 }

554 RunPosition backward_search_sub2(RunPosition rp, uint8_t c) const
555 {
556 uint8_t c_rp = this->dbwt.access_character_by_run_index(rp.run_index);
557
558 if (c == c_rp)
559 {
560 return rp;
561 }
562 else
563 {
564 uint64_t rank = this->dbwt.rank_on_first_characters_of_RLBWT(c, rp.run_index);
565 if (rank == 0)
566 {
567 return RunPosition::create_null_value();
568 }
569 else

578 BackwardSearchResult backward_search(const BackwardSearchResult &bsr, uint8_t c) const
579 {
580 RunPosition bx = this->dbwt.to_run_position(bsr.b);
581 RunPosition ex = this->dbwt.to_run_position(bsr.e);
582 uint8_t c_bx = this->dbwt.access_character_by_run_index(bx.run_index);
583 uint8_t c_ex = this->dbwt.access_character_by_run_index(ex.run_index);
584
585 if (c == c_bx)
586 {
587
588 int64_t next_b = this->dbwt.LF(bx);
589 int64_t next_sa_b_ = bsr.sa_b_ - 1;
590 int64_t next_e;
591 if (c_bx == c_ex)
592 {
593 next_e = this->dbwt.LF(ex);
594 }
595 else
596 {
597 RunPosition ey = this->backward_search_sub2(ex, c);
598 next_e = this->dbwt.LF(ey);
599 }
600 return BackwardSearchResult(next_b, next_e, next_sa_b_);
601 }
602 else
603 {
604
605 RunPosition by = this->backward_search_sub1(bx, c);
606 if (c == c_ex)
607 {
608
609 int64_t next_b = this->dbwt.LF(by);
610 int64_t next_e = this->dbwt.LF(ex);
611 int64_t next_sa_b_ = this->disa.get_sampled_first_sa_value(by.run_index) - 1;
612 if (next_sa_b_ == -1)
613 {
614 next_sa_b_ = this->text_size() - 1;
615 }
616 return BackwardSearchResult(next_b, next_e, next_sa_b_);
617 }
618 else
619 {
620 if (by.run_index <= ex.run_index)
621 {
622
623 RunPosition ey = this->backward_search_sub2(ex, c);
624 int64_t next_b = this->dbwt.LF(by);
625 int64_t next_e = this->dbwt.LF(ey);
626
627 int64_t next_sa_b_ = this->disa.get_sampled_first_sa_value(by.run_index) - 1;
628 if (next_sa_b_ == -1)
629 {
630 next_sa_b_ = this->text_size() - 1;
631 }
632
633 return BackwardSearchResult(next_b, next_e, next_sa_b_);
634 }
635 else
636 {
637 return BackwardSearchResult::create_empty_result();
638 }
639 }
640 }

649 BackwardSearchResult backward_search(const std::vector<uint8_t> &pattern) const
650 {
651 BackwardSearchResult tmp;
652 tmp.b = 0;
653 tmp.e = this->text_size() - 1;
654 tmp.sa_b_ = this->disa.get_sampled_first_sa_value(0);
655
656 for (int64_t i = pattern.size() - 1; i >= 0; i--)
657 {
658 tmp = this->backward_search(tmp, pattern[i]);
659 if (tmp.is_empty())
660 {
661 break;
662 }
663 }
664 return tmp;
665 }
672 std::vector<uint64_t> compute_sa_values(const BackwardSearchResult &bsr) const
673 {
674 std::vector<uint64_t> r;
675 if (!bsr.is_empty())
676 {
677 uint64_t x = bsr.sa_b_;
678 r.push_back(x);
679 for (int64_t i = bsr.b + 1; i <= bsr.e; i++)
680 {
681 x = this->disa.inverse_phi(x);
682 r.push_back(x);
683 }
684 }
685 return r;
