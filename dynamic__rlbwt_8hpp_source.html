493 {
494 if (nth == 0)
495 {
496 return -1;
497 }
498 else
499 {
500 nth--;
501 int64_t c_rank = this->cArray.get_c_id(c);
502 if (c_rank == -1)

504 return -1;
505 }
506 else
507 {

509 uint64_t psum1 = f_idx > 0 ? this->run_length_vector_sorted_by_F.psum(f_idx - 1) : 0;

511 uint64_t c_rank2 = x - f_idx;

520 uint64_t psum3 = nth3 > 0 ? this->run_length_vector.psum(nth3 - 1) : 0;
521 uint64_t c_rank3 = (psum1 + nth) - psum2;
