509 BackwardSearchResult backward_search(const Interval &intv, uint8_t c) const
510 {
511 uint64_t num1 = intv.first > 0 ? this->dbwt.rank(intv.first - 1, c) : 0;
512 uint64_t c_count = this->dbwt.get_c_array().get_c_count(c);
513 if (num1 < c_count)
514 {
515 uint64_t fst_c_pos = this->dbwt.select(num1, c);
516
517 if ((int64_t)fst_c_pos <= intv.second)
518 {
519 int64_t last_ith = this->dbwt.rank(intv.second, c);
520 uint64_t lst_c_pos = this->dbwt.select(last_ith-1, c);
521
522 return BackwardSearchResult(this->dbwt.LF(fst_c_pos), this->dbwt.LF(lst_c_pos));
523 }
524 else
525 {
526 return BackwardSearchResult::create_empty_result();
527 }
528 }
529 else
530 {
531 return BackwardSearchResult::create_empty_result();
532 }
533 }

542 BackwardSearchResult backward_search(const std::vector<uint8_t> &pattern) const
543 {
544 BackwardSearchResult bsr(0, this->size() - 1);
545 // Interval intv(0, this->size() - 1);
546 for (int64_t i = pattern.size() - 1; i >= 0; i--)
547 {
548 bsr = this->backward_search(bsr.get_sa_interval(), pattern[i]);
549 if (bsr.is_empty())

551 break;
552 }
553 }
554 return bsr;

721 uint64_t insert_char(int64_t pos, uint8_t c, FMIndexEditHistory *output_history = nullptr)
722 {
723 assert(c > this->dbwt.get_end_marker());
724 this->dbwt.verify_inserted_string({c});
725
726 SAIndex isa_of_insertionPosOnText = this->dsa.isa(pos);
727 // assert(isa_of_insertionPosOnText == this->dsa.isa(pos));
728
729 uint64_t positionToReplace = isa_of_insertionPosOnText;
730 uint64_t prev_isa = this->dbwt.LF(isa_of_insertionPosOnText);
731 uint64_t oldPositionToInsert = positionToReplace;
732
733 if (output_history != nullptr)
734 {
735 output_history->replaced_sa_index = positionToReplace;
736 output_history->type = EditType::InsertionOfChar;
737 output_history->inserted_string.push_back(c);
738 }
739
740 uint8_t oldChar = this->dbwt.access(positionToReplace);
741 uint64_t j = prev_isa;
742 uint8_t new_letter_L = c;
743
744 this->dbwt.set_character(positionToReplace, new_letter_L);
745 // this->dbwt.update_C_for_deletion(oldChar);
746 // this->dbwt.update_C_for_insertion(new_letter_L);
747
748 uint64_t C_count = this->dbwt.get_c_array().at(new_letter_L);
749 if (oldChar < new_letter_L)
750 {
751 C_count++;
752 }
753 uint64_t rank_value = this->dbwt.rank(positionToReplace, new_letter_L);
754 if (positionToReplace <= oldPositionToInsert && new_letter_L == oldChar)
755 {
756 rank_value++;
757 }
758
759 uint64_t positionToInsert = C_count + rank_value - 1;
760
761 if (output_history != nullptr)
762 {
763 output_history->inserted_sa_index = positionToInsert;
764 output_history->inserted_sa_indexes.push_back(positionToInsert);
765 }
766
767 this->dbwt.insert(positionToInsert, oldChar);
768 if (positionToInsert <= j)
769 {
770 j++;
771 }
772 if (positionToInsert <= positionToReplace)
773 {
774 positionToReplace++;
775 }
776
777 // this->disa.update_for_insertion(positionToInsert, pos);
778 this->dsa.update_for_insertion(positionToInsert, pos);
779
780 uint64_t j_prime = this->dbwt.LF(positionToInsert);
781
782 // std::cout << "j = " << j << ", j' = " << j_prime << std::endl;
783 if (output_history != nullptr)
784 {
785 output_history->first_j = j;
786 output_history->first_j_prime = j_prime;
787 }
788 std::vector<SAMove> swap_history = reorder_BWT(j, j_prime);
789 uint64_t sum = swap_history.size();
790
791 if (output_history != nullptr)
792 {
793 output_history->move_history.swap(swap_history);
794 }
795
796 this->dsa.update_sample_marks(pos + 1);
797
798 return sum;
799 }
